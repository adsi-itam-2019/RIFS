create table arrendatario(cArrendatario int primary key,
	telefono varchar(100) unique not null,
	passwd varchar(100) not null,
	nombre varchar(100) not null);

create table propietario(cPropietario int primary key,
	telefono varchar (100) unique not null,
	passwd varchar(100) not null,
	nombre varchar(100) not null);

create table propiedad (cPropiedad int primary key,
	direccion varchar(100) not null,
	amueblado bit,
	espacio real,
	roomies int,
	precio float not null,
	rentado bit not null,
	cPropietario int references propietario);

create table guarda(cPropiedad int references propiedad,
	cArrendatario int references arrendatario,
	primary key(cArrendatario,cPropiedad));

create table cita(cPropiedad int references propiedad,
	cArrendatario int references arrendatario,
	cPropietario int references propietario,
	fecha date,
	primary key(cArrendatario,cPropiedad,cPropietario));

create table Servicio(nombre varchar(100),
	cPropiedad int references propiedad,
	primary key (cPropiedad, nombre));
